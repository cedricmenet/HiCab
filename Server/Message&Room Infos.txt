	Message {
		"type" : string,
		"room" : string,
		"data" : CabInfos | CabLocalisation | CabRequests | CabRequest | RequestResponse
	}

	CabInfos {
		"IdCab" : string,
		"Odometer" : #,
		"IsBusy" : boolean,
		"Queue" : #
	}
	
	CabLocalisation {
		"IdCab" : string,
		"ActualPos" : LocVertex | LocStreet,
		"Destination" : LocVertex | LocStreet
	}
	
	CabRequests {
		"Requests": [CabRequest...]
	}
	
	CabRequest {
		"IdRequest" : string,
		"Position": LocVertex | LocStreet
	}
	
	RequestResponse {
		"Request" : CabRequest,
		"IdCab" : string,
		"IsAccepted" : boolean
	}
